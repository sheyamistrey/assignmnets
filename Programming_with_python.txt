#include<stdio.h>
int main()
{
    int n = 123456;
    int rem = 0;
    int sum = 0;
    while(n>0)
    {
        rem = n % 10;
        sum = sum + rem; //3
        n = n / 10;
        
    }
    printf("Sum of digits are : %d", sum);
    return 0;
}


#include<stdio.h>
int main()
{
    int n = 123456;
    int rem = 0;
    int rev = 0;
    while(n>0)
    {
        rem = n % 10;
        rev = rev*10 + rem; //3
        n = n / 10;
        
    }
    printf("Sum of digits are : %d", rev);
    return 0;
}
